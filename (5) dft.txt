close all; clc; clear all; 
X=input('enter the input sequence') 
N=length(X) 
for k=0:N-1 
     for n=0:N-1 
       p=exp(-j*2*pi*k*n/N);
      f(k+1,n+1)=p 
    end 
end 
X1=X*f 
figure(1)
 subplot(2,1,1)
 stem(abs(X1)) 
xlabel(' sample index') 
ylabel('amplitude')
 title('magnitude plot')
 subplot(2,1,2) 
stem(phase(X1)) 
xlabel(' sample index')
 ylabel('amplitude')
